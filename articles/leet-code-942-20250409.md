```go
func diStringMatch(s string) []int {
    n := len(s)
    result := make([]int, n+1)

    low, high := 0, n

    for i, c := range s {
        if c == 'I' {
            result[i] = low
            low++
        } else {
            result[i] = high
            high--
        }
    }
    result[n] = low // low == high

    return result
}
```
