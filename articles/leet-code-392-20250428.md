```go
func isSubsequence(s string, t string) bool {
    sPointer := 0
    sLength := len(s)

    if sLength == 0 {
        return true
    }

    for i := 0; i < len(t); i++ {
        if t[i] == s[sPointer] {
            sPointer++
            if sPointer == sLength {
                return true
            }
        }
    }

    return false
}
```

```go
func isSubsequence(s string, t string) bool {
    i, j := 0, 0

    for i < len(s) && j < len(t) {
        if s[i] == t[j] {
            i++
        }
        j++
    }

    return i == len(s)
}
```
