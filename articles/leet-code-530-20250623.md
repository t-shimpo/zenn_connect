func getMinimumDifference(root *TreeNode) int {
    var prev *int
    minDiff := math.MaxInt32

    var dfs func(node *TreeNode)
    dfs = func(node *TreeNode) {
        if node == nil {
            return
        }
        
        dfs(node.Left)

        if prev != nil {
            diff := node.Val - *prev
            if diff < minDiff {
                minDiff = diff
            }
        }
        prev = &node.Val

        dfs(node.Right)
    }
    
    dfs(root)
    return minDiff
}
