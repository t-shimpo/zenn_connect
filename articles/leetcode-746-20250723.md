```go
func minCostClimbingStairs(cost []int) int {
    prev2 := cost[0]
    prev1 := cost[1]

    for i := 2; i < len(cost); i++ {
        curr := min(prev1, prev2) + cost[i]
        prev2 = prev1
        prev1 = curr
    }

    return min(prev1, prev2)
}

func min(a,b int) int {
    if a < b {
        return a
    }
    return b
}
```
