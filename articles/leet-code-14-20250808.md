func longestCommonPrefix(strs []string) string {
	for i := 0; ; i++ {
		if i >= len(strs[0]) {
			return strs[0]
		}
		ch := strs[0][i]
		for j := 1; j < len(strs); j++ {
			if i >= len(strs[j]) || ch != strs[j][i] {
				return strs[0][:i]
			}
		}
	}
}
