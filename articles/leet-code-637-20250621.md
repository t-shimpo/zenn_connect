func averageOfLevels(root *TreeNode) []float64 {
    queue := []*TreeNode{root}
    var averages []float64

    for len(queue) > 0 {
        // queue ã«ã¯å¸¸ã«ã€Œç¾åœ¨ã®ãƒ¬ãƒ™ãƒ«ã®ãƒãƒ¼ãƒ‰ã€ãŒå…¥ã£ã¦ã„ã‚‹
        // ãã®ãƒãƒ¼ãƒ‰æ•°ã‚’å…ˆã«å–å¾—ã—ã¦ãŠãã“ã¨ã§ã€
        // æ¬¡ã®ãƒ¬ãƒ™ãƒ«ã®ãƒãƒ¼ãƒ‰ã‚’å¾Œã‚ã«è¿½åŠ ã—ã¤ã¤ã€ä»Šã®ãƒ¬ãƒ™ãƒ«ã®å‡¦ç†ãŒçµ‚ã‚ã‚‹ã“ã¨ã‚’ä¿è¨¼ã§ãã‚‹
        levelSize := len(queue)
        sum := 0

        for i := 0; i < levelSize; i++ {
            node := queue[0]
            queue = queue[1:]

            sum += node.Val

            if node.Left != nil {
                queue = append(queue, node.Left)          
            }
            if node.Right != nil {
                queue = append(queue, node.Right)      
            }
        }
        averages = append(averages, float64(sum) / float64(levelSize))
    }

    return averages
}
