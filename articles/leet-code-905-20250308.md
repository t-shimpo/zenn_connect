func sortArrayByParity(nums []int) []int {
	evens := []int{}
	odds := []int{}

	for _, num := range nums {
		if num%2 == 0 {
			evens = append(evens, num)
		} else {
			odds = append(odds, num)
		}
	}

	return append(evens, odds...)
}
