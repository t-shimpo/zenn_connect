```go
func isPalindrome(x int) bool {
	if x < 0 {
		return false
	}
	if x < 10 {
		return true
	}
	x_str := strconv.Itoa(x)
	left := 0
	right := len(x_str) - 1
	for left < right {
		if x_str[left] != x_str[right] {
			return false
		}
		left++
		right--
	}

	return true
}
```
