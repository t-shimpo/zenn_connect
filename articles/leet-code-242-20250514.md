```go
func isAnagram(s string, t string) bool {
	sLen := len(s)
	tLen := len(t)

	if sLen != tLen {
		return false
	}

	byteCount := make(map[byte]int)

	for i := range sLen {
		byteCount[s[i]]++
	}

	for j := range tLen {
		if byteCount[t[j]] == 0 {
			return false
		}
		byteCount[t[j]]--
	}

	return true
}
```

```go
func isAnagram(s string, t string) bool {
	sLen := len(s)
	tLen := len(t)

	if sLen != tLen {
		return false
	}

	byteCount := make([]int, 26)

	for i := range sLen {
		num := int(s[i] - 'a')
		byteCount[num]++
	}

	for j := range tLen {
		num := int(t[j] - 'a')
		if byteCount[num] == 0 {
			return false
		}
		byteCount[num]--
	}

	return true
}
```
